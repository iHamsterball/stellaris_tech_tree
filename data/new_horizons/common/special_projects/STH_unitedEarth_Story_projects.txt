special_project = {
	key = "XINDI_1_PROJECT"
	event_chain = "xindi_event_chain"
	cost = 0
	days_to_research = 30
	tech_department = engineering_technology
	picture = sth_GFX_evt_xindiProbe1
	timelimit = 1800
	event_scope = ship_event
	requirements = {
		shipclass_science_ship = 1
		leader = scientist
	}
	on_success = { 
		random_ambient_object = { limit = { has_ambient_object_flag = xindiProbeDebris } destroy_ambient_object = this }
		if = {
			limit = { exists = event_target:xindiProbeDebris }
			event_target:xindiProbeDebris = { destroy_ambient_object = this }
		}
		ship_event = { id = STH_united_earth_story.14 } 
	}
	on_fail = {	
		random_ambient_object = { limit = { has_ambient_object_flag = xindiProbeDebris } destroy_ambient_object = this }
		if = {
			limit = { exists = event_target:xindiProbeDebris }
			event_target:xindiProbeDebris = { destroy_ambient_object = this }
		}		
		capital_scope = { planet_event = { id = STH_united_earth_story.20 } }
		end_event_chain = "xindi_event_chain"
	}
}

special_project = {
	key = "XINDI_2_PROJECT"
	event_chain = "xindi_event_chain"
	cost = 0
	days_to_research = 30
	tech_department = engineering_technology
	picture = sth_GFX_evt_xindiProbe1
	timelimit = 1800
	event_scope = ship_event
	requirements = {
		shipclass_military = 1
		leader = admiral
	}
    on_success = {
        random_country = {
            limit = { has_country_flag = xindi_probe_target_country }
            save_event_target_as = xindi_probe_target_country
        }
        if = {
            limit = { exists = fleet }
            fleet = { fleet_event = { id = STH_united_earth_story.15 } }
		}
		else = {
			event_target:xindi_probe_target_country = {
				random_owned_fleet = {
					limit = { 
						exists = leader
						leader = { leader_class = admiral }
					}
					fleet_event = { id = STH_united_earth_story.15 }
				}
			}
		}
        event_target:xindi_probe_target_country = { end_event_chain = "xindi_event_chain" }
    }
	on_fail = { 
		capital_scope = { planet_event = { id = STH_united_earth_story.20 } } 
		end_event_chain = "xindi_event_chain"
	}
}
