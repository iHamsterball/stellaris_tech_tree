### Additional Triggers

is_peaceful_country = { 
	NOR = {
		has_ethic = ethic_fanatic_xenophobe
		has_ethic = ethic_fanatic_militarist
	}
}


valid_espionage_target = {
	NOT = { any_subject = { is_same_value = root } }
	NOT = { is_same_value = Root }
	is_assimilator = no
	is_infester = no
	is_normal_country = yes
	has_communications = Root
}

valid_diplomacy_espionage_target = {
	NOT = { any_subject = { is_same_value = root } }
	is_assimilator = no
	is_infester = no
	is_normal_country = yes
	has_communications = Root
	NOT = { is_same_value = Root }
	any_country = {
		is_assimilator = no
		is_infester = no
		is_normal_country = yes
		NOT = { is_same_value = Root }
		NOT = { is_same_value = prev }
		has_communications = prev
	}
}

valid_loan_target = {
	NOT = { has_country_flag = the_borg_collective }
	NOT = { has_country_flag = undine }
	#NOT = { has_country_flag = iconian_empire }
	is_normal_country = yes
	has_communications = Root
	NOT = { is_same_value = Root }
}


valid_privateer_target = {
	NOT = { any_subject = { is_same_value = root } }
	is_assimilator = no
	is_infester = no
	is_normal_country = yes
	has_communications = Root
	NOT = { is_same_value = Root }
}
