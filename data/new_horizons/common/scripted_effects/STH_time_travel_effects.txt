    		random_list = {

    				random_neighbor_country = {
    					limit = { 
    						is_country_type = default
    						num_owned_planets > 1
    					}
    					random_owned_planet = { 
    						limit = { 
    							is_capital = no 
    							is_homeworld = no 
								NOT = { has_planet_flag = "timeChanged" }
								solar_system = { 
									count_planets = { 
										limit = { is_colony = yes } 
										count = 1 
									} 
								}
    						}
    						set_timed_planet_flag = { flag = "timeChanged" days = 1800 }
    						set_controller = event_target:timelineCountry
							set_owner = event_target:timelineCountry
    						every_tile = {
    							limit = { exists = pop }
    							pop = { change_species = event_target:timelineCountrySpecies }
							}
							solar_system = {
								if = {
									limit = { exists = starbase }
									starbase = { set_owner = event_target:timelineCountry }
								}
							}
    					}
    				}
    			}
    		}
