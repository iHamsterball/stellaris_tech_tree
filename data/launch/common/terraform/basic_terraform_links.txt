# Arctic to Ocean
terraform_link = {
	from = "pc_arctic"
	to = "pc_ocean"
	
	#cost
	energy = 1500
	resources = {
		sr_terraform_gases = 2
		sr_terraform_liquids = 2
	}
	
	condition = {
		if = {
			limit = {
				OR = {
					ideal_planet_class = pc_arid
					ideal_planet_class = pc_desert
				}
			}
			has_technology = "tech_atmospheric_manipulation"
			else = {
				has_technology = "tech_terrestrial_sculpting"
			}
		}
	}
}
