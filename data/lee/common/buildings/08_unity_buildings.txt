building_autochthon_monument = {
	base_buildtime = 240
	base_cap_amount = 1

	category = unity
	
	potential = {
		exists = owner
		NOT = { has_modifier = slave_colony }
		owner = { 
			is_gestalt = no
			is_spiritualist = no
			is_megacorp = no
		}					
	}

		exists = owner
		OR = {
			has_modifier = slave_colony

				OR = {
					is_gestalt = yes

					is_megacorp = yes
				}
			}
		}
	}
	
	convert_to = {
		building_temple
		building_hive_node
		building_uplink_node
		building_corporate_monument
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}
	
	planet_modifier = {
		job_culture_worker_add = 2
	}
	
	prerequisites = {
		"tech_cultural_heritage"
	}	
	
	upgrades = {
		building_heritage_site
	}
	
	show_tech_unlock_if = {
		is_gestalt = no
		is_spiritualist = no
		is_megacorp = no
	}
	
	triggered_desc = {
		text = job_culture_worker_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			OR = {
				has_building = building_research_lab_1
				has_building = building_research_lab_2
				has_building = building_research_lab_3
				is_planet_class = pc_habitat
			}
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value > 10
				}
			}
		}
	}
}
