galactic_focus_pass_a_resolution = {
	icon = "GFX_galactic_focus_generic"

	effect = {
		add_resource = { influence = 100 }

		hidden_effect = {
			set_global_flag = galactic_focus_pass_a_resolution_completed
		}
	}

	potential = {
		NOT = { has_global_flag = galactic_focus_pass_a_resolution_completed }
	}

	trigger = {
		custom_tooltip = {
			fail_text = "galactic_community_resolution_passed"
			has_global_flag = galactic_community_resolution_passed
		}
	}
}
