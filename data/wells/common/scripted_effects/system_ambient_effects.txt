# Nebulae - Use in System Scope
create_nebula_1 = {
	star = {
		create_ambient_object = {
			type = "nebula_1"
			location = this
		}
		last_created_ambient_object = {
			set_location = {
				target = prev
				distance = 0
				angle = random
			}
		}
	}
}
