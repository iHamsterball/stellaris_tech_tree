is_pirate_system = {
	any_ship_in_system = {
		is_ship_size = pirate_station
	}
}

is_slavebot = { # The pop is a slave or level 1/2 robot
	OR = {
		is_enslaved = yes
		is_species = "ROBOT_POP_SPECIES_1"
		is_species = "ROBOT_POP_SPECIES_2"
	}
}

is_robot = { # The pop is a non-sentient robot
	OR = {
		is_species = "ROBOT_POP_SPECIES_1"
		is_species = "ROBOT_POP_SPECIES_2"
	}
}
